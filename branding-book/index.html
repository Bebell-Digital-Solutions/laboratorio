Conecta tu propio servidor gratuito usando Google Apps Script para guardar tu progreso.

Instrucciones:

1
Crea un nuevo proyecto en Apps Script.
2

Pega este código:

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    if (sheet.getLastRow() === 0) { sheet.appendRow(['Fecha', 'Usuario', 'Proyecto', 'Progreso', 'Datos del Brand (JSON)']); }
    sheet.appendRow([new Date(), data.params.user_id || 'Anónimo', data.project || 'Branding', data.params.progress || '0%', data.params.brand_data_json || '{}']);
    return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({success: false})).setMimeType(ContentService.MimeType.JSON);
  }
}
