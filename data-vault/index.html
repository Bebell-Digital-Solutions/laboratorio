1
Create a new Google Sheet. Name it anything you want (e.g. "Vault DB").
2
In the top menu of your sheet, click Extensions > Apps Script.
3
Paste the following code into the editor, replacing everything currently there: 
function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data = sheet.getDataRange().getValues();
  let results = [];
  if (data.length > 1) {
    const headers = data[0];
    for (let i = 1; i < data.length; i++) {
      let row = data[i];
      let obj = {};
      for (let j = 0; j < headers.length; j++) {
        obj[headers[j]] = headers[j] === 'tags' ? JSON.parse(row[j] || "[]") : row[j];
      }
      results.push(obj);
    }
  }
  return ContentService.createTextOutput(JSON.stringify(results))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const payload = JSON.parse(e.postData.contents);
  sheet.clear();
  if (!payload || payload.length === 0) return ContentService.createTextOutput("Success");
  
  const headers = Object.keys(payload[0]);
  sheet.appendRow(headers);
  const rows = payload.map(obj => headers.map(h => h === 'tags' ? JSON.stringify(obj[h]) : obj[h]));
  sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  
  return ContentService.createTextOutput(JSON.stringify({status: 'success'}))
    .setMimeType(ContentService.MimeType.JSON);
}
4
Click the blue Deploy button (top right) > New deployment.
5
Click the gear icon "Select type" and choose Web app. Under "Who has access", select Anyone. Click Deploy (Authorize access if prompted).
