            else {
                renderFilterBar();
                renderApp();
            }

            searchInput.addEventListener('input', (e) => {
                searchQuery = e.target.value.toLowerCase();
                renderApp();
            });

            btnGridView.addEventListener('click', () => setView('grid'));
            btnListView.addEventListener('click', () => setView('list'));

            // Tag Input Listeners
            tagInput.addEventListener('keydown', (e) => {
                if (e.key === 'Enter' || e.key === ',') {

            tagInput.addEventListener('blur', addTagFromInput);

        // --- Core Rendering ---
        function renderApp() {
            const filteredLinks = links.filter(link => {
                const searchString = `${link.name} ${link.desc} ${link.tags.join(' ')} ${link.url}`.toLowerCase();
                const matchesSearch = searchString.includes(searchQuery);
                const matchesFilter = activeTagFilter ? link.tags.includes(activeTagFilter) : true;
                return matchesSearch && matchesFilter;
            });

            if (links.length === 0) {
                emptyState.classList.remove('hidden');
                linksContainer.innerHTML = '';
                return;
            }

            emptyState.classList.add('hidden');

            if (filteredLinks.length === 0) {
                linksContainer.innerHTML = `
                    <div class="text-center py-20">
                        <div class="text-slate-400 mb-3"><svg class="w-12 h-12 mx-auto" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="1.5" d="M9.172 16.172a4 4 0 015.656 0M9 10h.01M15 10h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg></div>
                        <h3 class="text-lg font-medium text-slate-900">No results found</h3>
                        <p class="text-slate-500">Try adjusting your filters or search query.</p>
                    </div>`;
                return;
            }

            if (viewMode === 'grid') renderGrid(filteredLinks);
            else renderList(filteredLinks);
        }

        function renderFilterBar() {
            const allTags = new Set();
            links.forEach(l => l.tags.forEach(t => allTags.add(t)));
            const uniqueTags = Array.from(allTags).sort();

            if (uniqueTags.length === 0) {
                filterTagsContainer.classList.add('hidden');
                return;
            }

            filterTagsContainer.classList.remove('hidden');

            let html = `<button onclick="setFilter(null)" class="px-4 py-1.5 rounded-full text-sm font-medium whitespace-nowrap transition-colors flex-shrink-0 ${activeTagFilter === null ? 'bg-indigo-600 text-white shadow-sm' : 'bg-white border border-slate-200 text-slate-600 hover:bg-slate-50'}">All</button>`;

            uniqueTags.forEach(tag => {
                const isActive = activeTagFilter === tag;
                html += `<button onclick="setFilter('${tag}')" class="px-4 py-1.5 rounded-full text-sm font-medium whitespace-nowrap transition-colors flex-shrink-0 ${isActive ? 'bg-indigo-600 text-white shadow-sm' : 'bg-white border border-slate-200 text-slate-600 hover:bg-slate-50'}">${tag}</button>`;
            });

            filterTagsContainer.innerHTML = html;
        }

        function setFilter(tag) {
            activeTagFilter = tag;
            renderFilterBar();
            renderApp();
        }

        // --- Data Logic ---
        function saveToStorage() {
            localStorage.setItem('vault_links', JSON.stringify(links));
            updateMetrics();
            renderFilterBar();
            renderApp();
            
            // Sync to backend if configured
            if(gasUrl) syncToBackend();
        }

        function handleFormSubmit(e) {
            e.preventDefault();
            
            // Ensure any pending tag text is added
            addTagFromInput();

            const id = linkIdInput.value;
            const linkData = {
                id: id || Date.now().toString(),
                url: linkUrlInput.value,
                name: linkNameInput.value,
                desc: linkDescInput.value,
                tags: [...currentTags],
                updatedAt: new Date().toISOString()
            };

            let isNew = false;
            if (id) {
                const index = links.findIndex(l => l.id === id);
                if (index > -1) links[index] = linkData;
                showToast('Link updated successfully');
            } else {
                links.unshift(linkData);
                isNew = true;
                showToast('Link added successfully');
            }

            saveToStorage();
            closeAddModal();

            if (isNew) {
                fireConfetti();
            }
        }

        function editLink(id, event) {
            if(event) event.stopPropagation();
            const link = links.find(l => l.id === id);
            if (!link) return;

            linkIdInput.value = link.id;
            linkUrlInput.value = link.url;
            linkNameInput.value = link.name;
            linkDescInput.value = link.desc;
            currentTags = [...link.tags];
            
            formTitle.textContent = 'Edit Link';
            renderTags();
            
            modalAdd.classList.remove('hidden');
        }

        function deleteLink(id, event) {
            if(event) event.stopPropagation();
            if (confirm("Are you sure you want to remove this link?")) {
                links = links.filter(l => l.id !== id);
                saveToStorage();
                showToast('Link deleted');
            }
        }

        // --- Animations & UI Details ---
        function updateMetrics() {
            const endValue = links.length;
            animateValue("totalCount", animatedCount, endValue, 600);
            animatedCount = endValue;
        }

        function animateValue(id, start, end, duration) {
            if (start === end) return;
            let obj = document.getElementById(id);
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                obj.innerHTML = Math.floor(progress * (end - start) + start);
                if (progress < 1) {
                    window.requestAnimationFrame(step);
                } else {
                    // Small pop effect when animation finishes
                    obj.classList.add('scale-125', 'transition-transform');
                    setTimeout(() => obj.classList.remove('scale-125'), 150);
                }
            };
            window.requestAnimationFrame(step);
        }

        function fireConfetti() {
            if(typeof confetti === 'function') {
                confetti({
                    particleCount: 100,
                    spread: 70,
                    origin: { y: 0.6 },
                    colors: ['#4f46e5', '#818cf8', '#c7d2fe', '#f8fafc']
                });
            }
        }

        // --- Backend Sync (Google Apps Script API logic) ---
        async function fetchFromBackend() {
            try {
                const res = await fetch(gasUrl, { method: 'GET' });
                if(res.ok) {
                    const data = await res.json();
                    if(Array.isArray(data)) {
                        links = data;
                        localStorage.setItem('vault_links', JSON.stringify(links));
                    }
                }
            } catch (error) {
                console.warn("Could not fetch from backend. Using local storage.", error);
            }
            updateMetrics();
            renderFilterBar();
            renderApp();
        }

        async function syncToBackend() {
            try {
                await fetch(gasUrl, {
                    method: 'POST',
                    body: JSON.stringify(links),
                    headers: { 'Content-Type': 'application/json' }
                });
            } catch (error) {
                console.error("Failed to sync to backend. Data is saved locally.", error);
            }
        }

        function saveSettings() {
            const url = settingsGasUrl.value.trim();
            localStorage.setItem('vault_gas_url', url);
            gasUrl = url;
            closeSettingsModal();
            showToast('Settings saved');
            if(gasUrl) fetchFromBackend();
        }

        // --- Tag Management ---
        function renderTags() {
            tagsContainer.innerHTML = currentTags.map((tag, index) => `
                <span class="bg-indigo-100 text-indigo-700 text-sm px-2 py-1 rounded-md flex items-center gap-1 group">
                    ${tag}
                    <button type="button" onclick="removeTag(${index})" class="text-indigo-400 hover:text-indigo-600 focus:outline-none opacity-50 group-hover:opacity-100">
                        <svg class="w-3 h-3" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path></svg>
                    </button>
                </span>
            `).join('');
        }

        function addTagFromInput() {
            const value = tagInput.value.trim().replace(',', '');
            if (value && !currentTags.includes(value)) {
                currentTags.push(value);
                renderTags();
            }
            tagInput.value = '';
        }

        function removeTag(index) {
            currentTags.splice(index, 1);
            renderTags();
        }

        // --- UI Interactions ---
        function setView(view) {
            viewMode = view;
            localStorage.setItem('vault_view', view);
            updateViewToggleButtons();
            renderApp();
        }

        function updateViewToggleButtons() {
            if (viewMode === 'grid') {
                btnGridView.classList.add('bg-white', 'shadow-sm', 'text-slate-900');
                btnGridView.classList.remove('text-slate-500');
                btnListView.classList.remove('bg-white', 'shadow-sm', 'text-slate-900');
                btnListView.classList.add('text-slate-500');
            } else {
                btnListView.classList.add('bg-white', 'shadow-sm', 'text-slate-900');
                btnListView.classList.remove('text-slate-500');
                btnGridView.classList.remove('bg-white', 'shadow-sm', 'text-slate-900');
                btnGridView.classList.add('text-slate-500');
            }
        }

        function openAddModal() {
            document.getElementById('linkForm').reset();
            linkIdInput.value = '';
            currentTags = [];
            renderTags();
            formTitle.textContent = 'Save a Link';
            modalAdd.classList.remove('hidden');
            setTimeout(() => linkUrlInput.focus(), 100);
        }

        function closeAddModal() {
            modalAdd.classList.add('hidden');
        }

        function openSettingsModal() {
            settingsGasUrl.value = gasUrl;
            modalSettings.classList.remove('hidden');
        }

        function closeSettingsModal() {
            modalSettings.classList.add('hidden');
        }

        function openUrl(url) {
            window.open(url, '_blank', 'noopener,noreferrer');
        }

        function copyUrl(event, url) {
            event.stopPropagation();
            if (navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(url).then(() => showToast('URL copied to clipboard!'));
            } else {
                const textArea = document.createElement("textarea");
                textArea.value = url;
                textArea.style.position = "fixed";
                textArea.style.left = "-999999px";
                textArea.style.top = "-999999px";
                document.body.appendChild(textArea);
                textArea.focus();
                textArea.select();
                try {
                    document.execCommand('copy');
                    showToast('URL copied to clipboard!');
                } catch (error) {
                    console.error('Fallback copy failed', error);
                }
                textArea.remove();
            }
        }

        function copyGasCode(event) {
            event.stopPropagation();
            const code = document.getElementById('gasCodeBlock').textContent;
            if (navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(code).then(() => showToast('Script code copied!'));
            } else {
                const textArea = document.createElement("textarea");
                textArea.value = code;
                textArea.style.position = "fixed";
                textArea.style.left = "-999999px";
                textArea.style.top = "-999999px";
                document.body.appendChild(textArea);
                textArea.focus();
                textArea.select();
                try {
                    document.execCommand('copy');
                    showToast('Script code copied!');
                } catch (error) {
                    console.error('Fallback copy failed', error);
                }
                textArea.remove();
            }
        }

        function showToast(message) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMessage').textContent = message;
            
            toast.classList.remove('translate-y-20', 'opacity-0');
            toast.classList.add('translate-y-0', 'opacity-100');
            
            setTimeout(() => {
                toast.classList.remove('translate-y-0', 'opacity-100');
                toast.classList.add('translate-y-20', 'opacity-0');
            }, 3000);
        }

        // Utility
        function getDomain(url) {
            try {
                return new URL(url).hostname.replace('www.', '');
            } catch (e) {
                return url;
            }
        }

        // Link metadata autofill utility
        linkUrlInput.addEventListener('blur', function() {
            if(!linkNameInput.value && this.value) {
                try {
                    let domain = new URL(this.value).hostname.replace('www.', '');
                    let parts = domain.split('.');
                    if(parts.length > 1) parts.pop(); 
                    let name = parts.join(' ');
                    linkNameInput.value = name.charAt(0).toUpperCase() + name.slice(1);
                } catch(e) {}
            }
        });
